        tensorflow=2.0.0（若有GPU则tensorflow-gpu=2.0.0）
        opencv=3.4.2
        django=2.2.5
        pandas=0.25.3
        pymysql==0.9.3
        requests=2.22.0
        scikit-learn=0.21.3
        tqdm=4.38.0
        ujson=1.35

        image==1.5.27
        gunicorn==20.0.4
        oss2==2.9.0
        boto3==1.11.15
